---
title: 'Retrieve Product'
api: 'GET /api/v1/products/{id}'
description: 'Retrieve a specific product by ID'
---

## Request

<ParamField path="id" type="string" required>
  The product ID (CzUid) to retrieve
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="data" type="object">
  The product object
  
  <Expandable title="Product Object">
    <ResponseField name="id" type="string">
      Unique product identifier (CzUid)
    </ResponseField>
    
    <ResponseField name="object" type="string">
      Always `product`
    </ResponseField>
    
    <ResponseField name="name" type="string">
      Product name
    </ResponseField>
    
    <ResponseField name="description" type="string">
      Product description
    </ResponseField>
    
    <ResponseField name="currency" type="string">
      Currency code
    </ResponseField>
    
    <ResponseField name="price" type="number">
      Selling price (without VAT)
    </ResponseField>
    
    <ResponseField name="vat_rate" type="number">
      VAT rate percentage
    </ResponseField>
    
    <ResponseField name="unit_of_measure" type="string">
      Unit of measure
    </ResponseField>
    
    <ResponseField name="ubl_um" type="string">
      UBL unit of measure code
    </ResponseField>
    
    <ResponseField name="ubl_nc" type="string">
      Combined Nomenclature code
    </ResponseField>
    
    <ResponseField name="ubl_cpv" type="string">
      CPV code
    </ResponseField>
    
    <ResponseField name="created_at" type="integer">
      Unix timestamp of creation
    </ResponseField>
    
    <ResponseField name="modified_at" type="integer">
      Unix timestamp of last update
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
  Response metadata including version and response time
</ResponseField>

<RequestExample>

```bash cURL
curl https://app.contazen.ro/api/v1/products/prod_5f6g7h8i9j \
  -H "Authorization: Bearer sk_live_YOUR_API_KEY"
```

```javascript Node.js
const response = await fetch('https://app.contazen.ro/api/v1/products/prod_5f6g7h8i9j', {
  headers: {
    'Authorization': 'Bearer sk_live_YOUR_API_KEY'
  }
});

const data = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://app.contazen.ro/api/v1/products/prod_5f6g7h8i9j',
    headers={'Authorization': 'Bearer sk_live_YOUR_API_KEY'}
)

data = response.json()
```

```php PHP
$ch = curl_init('https://app.contazen.ro/api/v1/products/prod_5f6g7h8i9j');
curl_setopt($ch, CURLOPT_HTTPHEADER, [
    'Authorization: Bearer sk_live_YOUR_API_KEY'
]);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);

$response = curl_exec($ch);
$data = json_decode($response, true);
```

## Notes

- The product data structure is identical to what's returned in the list endpoint
- All fields are returned for individual product retrieval
- Products are firm-specific - you can only retrieve products belonging to your authenticated firm
