---
title: 'Mark Invoice Paid'
api: 'POST /api/v1/invoices/{id}/pay'
description: 'Record a payment for an invoice'
---

## Request

<ParamField path="id" type="string" required>
  The invoice ID (CzUid) to mark as paid
</ParamField>

<ParamField body="amount" type="number">
  Payment amount. If not specified, the full invoice amount will be used.
</ParamField>

<ParamField body="payment_date" type="string">
  Payment date in YYYY-MM-DD format. Defaults to today if not specified.
</ParamField>

<ParamField body="payment_method" type="string" default="bank_transfer">
  Payment method. Available options:
  - `receipt` - Receipt
  - `bank_transfer` - Bank transfer (default)
  - `card` - Credit/debit card
  - `check` - Check
  - `promissory_note` - Promissory note
  - `fiscal_receipt` - Fiscal receipt
  - `other` - Other payment method
</ParamField>

<ParamField body="observations" type="string">
  Optional notes or observations about the payment
</ParamField>

<Note>
  Recording a payment for the full invoice amount will automatically mark the invoice as paid (`is_paid = true`).
</Note>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="Properties">
    <ResponseField name="payment_id" type="string">
      The created payment ID
    </ResponseField>
    
    <ResponseField name="invoice_id" type="string">
      The invoice CzUid
    </ResponseField>
    
    <ResponseField name="amount" type="number">
      The payment amount
    </ResponseField>
    
    <ResponseField name="payment_date" type="string">
      The payment date
    </ResponseField>
    
    <ResponseField name="payment_method" type="string">
      The payment method used
    </ResponseField>
    
    <ResponseField name="is_paid" type="boolean">
      Whether the invoice is now fully paid
    </ResponseField>
    
    <ResponseField name="remaining_amount" type="number">
      Remaining amount to be paid (0 if fully paid)
    </ResponseField>
    
    <ResponseField name="message" type="string">
      Success message
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
  Response metadata including version and response time
</ResponseField>

]);

$response = curl_exec($ch);
$data = json_decode($response, true);
