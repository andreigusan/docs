---
title: 'List Invoices'
api: 'GET /api/v1/invoices'
description: 'Retrieve a paginated list of invoices'
---

## Request

<ParamField query="page" type="integer" default="1">
  Page number to retrieve
</ParamField>

<ParamField query="per_page" type="integer" default="50">
  Number of items per page (max: 100)
</ParamField>

<ParamField query="client_id" type="string">
  Filter by client CzUid
</ParamField>

<ParamField query="document_type" type="string">
  Filter by document type
  
  Options:
  - `fiscal` - Fiscal invoice
  - `proforma` - Proforma invoice
  - `receipt` - Receipt
</ParamField>

<ParamField query="is_paid" type="integer">
  Filter by payment status
  
  Options:
  - `0` - Unpaid invoices
  - `1` - Paid invoices
</ParamField>

<ParamField query="start_date" type="string">
  Filter by start date (format: YYYY-MM-DD)
</ParamField>

<ParamField query="end_date" type="string">
  Filter by end date (format: YYYY-MM-DD)
</ParamField>

<ParamField query="sort" type="string" default="created_at">
  Field to sort by
  
  Options:
  - `created_at` - Creation date
  - `due_date` - Due date
  - `total` - Total amount
</ParamField>

<ParamField query="order" type="string" default="desc">
  Sort order. Options: `asc`, `desc`
</ParamField>

<ParamField query="expand[]" type="array">
  Expand nested objects. See [Expanding Objects](/concepts/expanding-objects) for details.
  
  Available expansions:
  - `lines` - Invoice line items
  - `payments` - Payment records
  - `client` - Client details
  - `efactura` - E-Factura submission info
</ParamField>

<RequestExample>

```bash cURL
curl https://app.contazen.ro/api/v1/invoices \
  -H "Authorization: Bearer sk_live_YOUR_API_KEY"
```

```bash cURL with Filters
curl https://app.contazen.ro/api/v1/invoices \
  -H "Authorization: Bearer sk_live_YOUR_API_KEY" \
  -G \
  -d document_type=fiscal \
  -d is_paid=0 \
  -d start_date=2024-01-01 \
  -d end_date=2024-12-31 \
  -d "expand[]=client"
```

```javascript Node.js
const response = await fetch('https://app.contazen.ro/api/v1/invoices?' + 
  new URLSearchParams({
    document_type: 'fiscal',
    is_paid: '0',
    page: '1',
    per_page: '50'
  }), {
  headers: {
    'Authorization': 'Bearer sk_live_YOUR_API_KEY'
  }
});

const data = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://app.contazen.ro/api/v1/invoices',
    params={
        'document_type': 'fiscal',
        'is_paid': 0,
        'expand': ['client', 'lines']
    },
    headers={'Authorization': 'Bearer sk_live_YOUR_API_KEY'}
)

data = response.json()
```

```php PHP
$ch = curl_init();
$params = http_build_query([
    'document_type' => 'fiscal',
    'is_paid' => 0,
    'page' => 1
]);

curl_setopt($ch, CURLOPT_URL, 'https://app.contazen.ro/api/v1/invoices?' . $params);
curl_setopt($ch, CURLOPT_HTTPHEADER, [
    'Authorization: Bearer sk_live_YOUR_API_KEY'
]);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);

$response = curl_exec($ch);
$data = json_decode($response, true);
```
