---
title: 'Update Client'
api: 'PUT /api/v1/clients/{id}'
description: 'Update an existing client'
---

## Request

<ParamField path="id" type="string" required>
  The client ID to update
</ParamField>

<ParamField body="name" type="string">
  Client name or company name
</ParamField>

<ParamField body="email" type="string">
  Email address
</ParamField>

<ParamField body="phone" type="string">
  Phone number
</ParamField>

<ParamField body="address" type="string">
  Street address
</ParamField>

<ParamField body="city" type="string">
  City name. For București, include sector (e.g., "București, Sectorul 1")
  
  <Note>
    **E-Factura Requirement**: For Romanian e-invoicing compliance, București addresses must include the sector in the format "București, Sectorul X"
  </Note>
</ParamField>

<ParamField body="county" type="string">
  County or state
</ParamField>

<ParamField body="postal_code" type="string">
  Postal or ZIP code
</ParamField>

<ParamField body="iban" type="string">
  Bank account IBAN
</ParamField>

<ParamField body="bank" type="string">
  Bank name
</ParamField>

<ParamField body="contact_person" type="string">
  Contact person name
</ParamField>

<Note>
  Only fields provided in the request body will be updated. Fields not included will remain unchanged.
</Note>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="data" type="object">
  The updated client object
</ResponseField>

<ResponseField name="meta" type="object">
  Response metadata
</ResponseField>

```python Python
import requests

def update_client(client_id):
    url = f'https://app.contazen.ro/api/v1/clients/{client_id}'
    headers = {
        'Authorization': 'Bearer sk_live_YOUR_API_KEY',
        'Content-Type': 'application/json'
    }
    
    data = {
        'phone': '+40722999888',
        'address': 'Bd. Tehnologie nr. 100',
        'city': 'București, Sectorul 2',
        'contact_person': 'Maria Ionescu'
    }
    
    response = requests.put(url, json=data, headers=headers)
    return response.json()
```

```php PHP
<?php
$clientId = '9z8y7x6w5v';

    'Authorization: Bearer sk_live_YOUR_API_KEY',
    'Content-Type: application/json'
]);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);

$response = curl_exec($ch);
$result = json_decode($response, true);
?>
```
