title: 'Create Client'

description: 'Create a new client'
---

## Request

<ParamField body="name" type="string" required>
  Client name or company name
</ParamField>

<ParamField body="email" type="string">
  Email address
</ParamField>

<ParamField body="cui" type="string">
  Tax ID (CUI/CIF) - required for B2B clients
</ParamField>

<ParamField body="cui_prefix" type="string" default="RO">
  CUI prefix for international companies
</ParamField>

<ParamField body="rc" type="string">
  Registration code (e.g., J40/1234/2023)
</ParamField>

<ParamField body="phone" type="string">
  Phone number
</ParamField>

<ParamField body="address" type="string">
  Street address
</ParamField>

<ParamField body="city" type="string">
  City name. For București, include sector (e.g., "București, Sectorul 1")
  
  <Note>
    **E-Factura Requirement**: For Romanian e-invoicing compliance, București addresses must include the sector in the format "București, Sectorul X"
  </Note>
</ParamField>

<ParamField body="county" type="string">
  County or state
</ParamField>

<ParamField body="country" type="string" default="RO">
  Country code (ISO 3166-1 alpha-2)
</ParamField>

<ParamField body="postal_code" type="string">
  Postal or ZIP code
</ParamField>

<ParamField body="iban" type="string">
  Bank account IBAN
</ParamField>

<ParamField body="bank" type="string">
  Bank name
</ParamField>

<ParamField body="contact_person" type="string">
  Contact person name
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="data" type="object">
  The created client object
</ResponseField>

<ResponseField name="meta" type="object">
  Response metadata
</ResponseField>

import requests

def create_client():
    url = 'https://app.contazen.ro/api/v1/clients'
    headers = {
        'Authorization': 'Bearer sk_live_YOUR_API_KEY',
        'Content-Type': 'application/json'
    }
    
    data = {
        'name': 'Tech Solutions SRL',
        'email': 'contact@techsolutions.ro',
        'cui': 'RO98765432',

        'phone': '+40722334455',
        'address': 'Bd. Tehnologie nr. 45',
        'city': 'București, Sectorul 2',
        'county': 'București',
        'country': 'RO',
        'postal_code': '020202',
        'iban': 'RO49AAAA1B31007593840000',
        'bank': 'BCR'
    }
    
    response = requests.post(url, json=data, headers=headers)
